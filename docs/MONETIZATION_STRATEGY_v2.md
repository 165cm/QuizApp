```javascript
// pages/quiz/[category]/[topic]/page.tsx
export async function generateStaticParams() {
  const topics = await getTopicsFromDB(); // 1,000トピック
  return topics.map(t => ({
    category: t.category,
    topic: t.slug
  }));
}

export default async function QuizPage({ params }) {
  const quiz = await getPreGeneratedQuiz(params.topic);

  return (
    <div>
      <h1>{quiz.title} AI練習クイズ</h1>
      <SampleQuiz questions={quiz.sampleQuestions} />
      <CTAButton>無料で全問題を解く</CTAButton>
      <RelatedTopics category={params.category} />
    </div>
  );
}
```
